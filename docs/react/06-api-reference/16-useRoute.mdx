---
title: useRoute
description: Read typed route params for the current page.
---

Returns the typed params for the current route. The type parameter should be the `PageProps` type generated for the current route file.

```tsx
import type { PageProps } from './$types'
import { useRoute } from '$houdini'

export function ShowBreadcrumb() {
  const { params } = useRoute<PageProps>()
  return <span>Show #{params.id}</span>
}
```

**Signature**

```ts
function useRoute<PageProps>(): { params: PageProps['Params'] }
```
